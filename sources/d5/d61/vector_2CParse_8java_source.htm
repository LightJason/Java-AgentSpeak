75  @Nonnull

77  public final IFuzzyValue<Boolean> execute( final boolean p_parallel, @Nonnull final IContext p_context,
78  @Nonnull final List<ITerm> p_argument, @Nonnull final List<ITerm> p_return )
79  {
80  final List<ITerm> l_arguments = CCommon.flatten( p_argument ).collect( Collectors.toList() );
81  final int l_limit;
82  final EType l_type;
83  if ( ( CCommon.rawvalueAssignableTo( l_arguments.get( l_arguments.size() - 1 ), String.class ) )
84  && ( EType.exists( l_arguments.get( l_arguments.size() - 1 ).<String>raw() ) ) )
85  {
86  l_type = EType.from( l_arguments.get( l_arguments.size() - 1 ).<String>raw() );
87  l_limit = l_arguments.size() - 1;
88  }
89  else
90  {
91  l_type = EType.DENSE;
92  l_limit = l_arguments.size();
93  }
94 
95 
96  // create vectors
97  switch ( l_type )
98  {
99  case DENSE:
100  l_arguments.stream()
101  .limit( l_limit )
102  .map( ITerm::<String>raw )
103  .map( CParse::parse )
104  .map( DenseDoubleMatrix1D::new )
105  .map( CRawTerm::from )
106  .forEach( p_return::add );
107 
108  return CFuzzyValue.from( true );
109 
110  case SPARSE:
111  l_arguments.stream()
112  .limit( l_limit )
113  .map( ITerm::<String>raw )
114  .map( CParse::parse )
115  .map( SparseDoubleMatrix1D::new )
116  .map( CRawTerm::from )
117  .forEach( p_return::add );
118 
119  return CFuzzyValue.from( true );
120 
121  default:
122  }
123 
124  return CFuzzyValue.from( false );
125  }
